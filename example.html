Exemplo de uso da iteração sobre estados

#include "gov_dev.h"
int main()
{
int i;
reset_state_counter();
for(i = 0; i < 30, i++)
{
printf("Estado atual: %s\n", current_state_name());
next_state();
}
wait_enter("Tecle ENTER para continuar...");
show_manufacturers_by_id();
return 0;
}
